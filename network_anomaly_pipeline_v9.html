# Utilities
def metrics_from_cm(cm):
    tn, fp, fn, tp = cm.ravel()
    acc = (tp+tn)/(tp+tn+fp+fn)
    tpr = tp/(tp+fn) if (tp+fn)>0 else 0.0
    fpr = fp/(fp+tn) if (fp+tn)>0 else 0.0
    return acc, tpr, fpr

def evaluate_predictions(y_true, y_pred, title=""):
    cm = confusion_matrix(y_true, y_pred, labels=[0,1])
    acc, tpr, fpr = metrics_from_cm(cm)
    print(f"{title} -> Accuracy: {acc:.4f}, TPR (PD): {tpr:.4f}, FPR (PFA): {fpr:.4f}")
    print("Confusion matrix [ [TN FP],[FN TP] ]\n", cm)
    return {'accuracy':acc, 'tpr':tpr, 'fpr':fpr, 'cm':cm.tolist()}

def sid_to_order(s):
    # session_id like 'SID_00001' -> 1
    try:
        return int(str(s).split('_')[-1])
    except:
        return 0

def make_time_split(df_local, test_ratio=0.3):
    order = df_local['session_id'].map(sid_to_order)
    idx = order.sort_values().index
    n = len(idx)
    split = int(math.floor((1-test_ratio)*n))
    train_idx = idx[:split]
    test_idx = idx[split:]
    return train_idx, test_idx

def make_host_disjoint_split(df_local, test_hosts=("Edge","Unknown")):
    # Train on all except test_hosts in browser_type
    test_mask = df_local['browser_type'].isin(list(test_hosts))
    train_idx = df_local.index[~test_mask]
    test_idx  = df_local.index[test_mask]
    return train_idx, test_idx

def export_predictions_csv(path, df_ids, y_true, y_pred, y_score=None):
    out = pd.DataFrame({'session_id': df_ids, 'y_true': y_true, 'y_pred': y_pred})
    if y_score is not None:
        out['score'] = y_score
    out.to_csv(path, index=False)
    return out

def cm_counts_and_perc(y_true, y_pred):
    """
    Returns:
      counts: 2x2 array [[TN, FP],[FN, TP]]
      perc:   2x2 array as fractions of the TOTAL (for percent heatmap)
    """
    cm = confusion_matrix(np.asarray(y_true).astype(int),
                          np.asarray(y_pred).astype(int))
    cm = cm.astype(int)
    total = cm.sum()
    perc = cm / total if total > 0 else np.zeros_like(cm, dtype=float)
    return cm, perc

def plot_cm_heatmap(cm_counts, cm_perc, title):
    """
    Draws a seaborn heatmap of percentages with quadrant labels + counts.
    annot=True, fmt='.2%', cmap='Blues' as requested.
    """
    fig, ax = plt.subplots(figsize=(6, 5.4))
    sns.heatmap(
        cm_perc,
        annot=True, fmt='.2%', cmap='Blues',
        cbar=True, square=True,
        xticklabels=['Predicted 0','Predicted 1'],
        yticklabels=['Actual 0','Actual 1'],
        ax=ax
    )
    ax.set_title(title)

    # Overlay quadrant labels + counts in each cell
    labels = np.array([["True Neg", "False Pos"], ["False Neg", "True Pos"]])
    counts = cm_counts

    # place two extra lines of text in each cell: label near top, count near bottom
    for i in range(2):
        for j in range(2):
            ax.text(j + 0.5, i + 0.20, labels[i, j],
                    ha='center', va='center', fontsize=10, color='black', fontweight='bold')
            ax.text(j + 0.5, i + 0.80, f"n={counts[i, j]:,}",
                    ha='center', va='center', fontsize=10, color='black')

    plt.tight_layout()
    plt.show()

def _scores_to_preds(scores, threshold=0.5):
    """Convert continuous scores/probabilities to 0/1 labels at a threshold."""
    s = np.asarray(scores, dtype=float)
    return (s >= float(threshold)).astype(int)

# ===== Plots: Compare TPR/FPR/ACC across models for both splits =====
def plot_bars(results_dict, title):
    labs = list(results_dict.keys())
    acc = [results_dict[k]['accuracy'] for k in labs]
    tpr = [results_dict[k]['tpr'] for k in labs]
    fpr = [results_dict[k]['fpr'] for k in labs]
    x = np.arange(len(labs))
    w = 0.25

def export_feature_importance(pipe, X_fit, y_fit, path_csv):
    import numpy as np, pandas as pd
    from sklearn.inspection import permutation_importance

    prep = pipe.named_steps['prep']
    model = pipe.named_steps['model']

    # Try transformed feature names
    feat_names = None
    try:
        feat_names = prep.get_feature_names_out()
    except Exception:
        try:
            ohe = prep.named_transformers_['cat']
            num_cols = prep.transformers_[0][2]
            cat_cols = prep.transformers_[1][2]
            feat_names = np.array(list(num_cols) + list(ohe.get_feature_names_out(cat_cols)))
        except Exception:
            pass

    imps = getattr(model, 'feature_importances_', None)
    if (imps is not None) and (feat_names is not None) and (len(imps) == len(feat_names)):
        imp_df = pd.DataFrame({'feature': feat_names, 'importance': np.asarray(imps)})
    elif (imps is not None) and (feat_names is not None):
        m = min(len(feat_names), len(imps))
        imp_df = pd.DataFrame({'feature': feat_names[:m], 'importance': np.asarray(imps)[:m]})
    elif imps is not None:
        imp_df = pd.DataFrame({'feature': [f'f{i}' for i in range(len(imps))], 'importance': np.asarray(imps)})
    else:
        # Fallback: permutation importance in original space
        r = permutation_importance(pipe, X_fit, y_fit, n_repeats=5, random_state=SEED, n_jobs=-1, scoring='f1')
        base_names = list(X_fit.columns)
        m = min(len(base_names), len(r.importances_mean))
        imp_df = pd.DataFrame({'feature': base_names[:m], 'importance': r.importances_mean[:m]})

    imp_df.sort_values('importance', ascending=False, inplace=True)
    imp_df.to_csv(path_csv, index=False)
    return imp_df

def bootstrap_auc_ci(y_true, y_scores, n_bootstrap=1000):
    aucs = []
    for _ in range(n_bootstrap):

# Better bootstrap: stratified + seeded 
def bootstrap_auc_ci_stratified(y_true, y_scores, n_bootstrap=2000, ci=0.95, random_state=42):
    y_true = np.asarray(y_true).astype(int)
    y_scores = np.asarray(y_scores).astype(float)
    base_auc = roc_auc_score(y_true, y_scores)

    rng = np.random.default_rng(random_state)
    pos = np.where(y_true == 1)[0]
    neg = np.where(y_true == 0)[0]
    n_pos, n_neg = len(pos), len(neg)
    if n_pos == 0 or n_neg == 0:
        raise ValueError("Need at least one positive and one negative for AUC.")

    aucs = np.empty(n_bootstrap, dtype=float)
    for b in range(n_bootstrap):
        samp_pos = rng.choice(pos, size=n_pos, replace=True)
        samp_neg = rng.choice(neg, size=n_neg, replace=True)
        idx = np.concatenate([samp_pos, samp_neg])
        aucs[b] = roc_auc_score(y_true[idx], y_scores[idx])

    alpha = (1 - ci) / 2
    lo, hi = np.quantile(aucs, [alpha, 1-alpha])
    return base_auc, lo, hi

# Low-FPR utilities 
def tpr_at_fpr_targets(y_true, y_scores, fpr_targets=(0.001, 0.002, 0.005, 0.01)):
    fpr, tpr, _ = roc_curve(y_true, y_scores)
    order = np.argsort(fpr)
    fpr, tpr = fpr[order], tpr[order]
    out = {}
    for tgt in fpr_targets:
        if tgt <= fpr[0]: out[tgt] = float(tpr[0]); continue
        if tgt >= fpr[-1]: out[tgt] = float(tpr[-1]); continue
        i = np.searchsorted(fpr, tgt, side="right")
        x0, x1 = fpr[i-1], fpr[i]
        y0, y1 = tpr[i-1], tpr[i]
        out[tgt] = float(y0 if x1==x0 else y0 + (y1-y0)*(tgt-x0)/(x1-x0))
    return out

def ppv_from_fpr_tpr(prevalence, fpr, tpr):
    # PPV = (tpr * ) / (tpr* + fpr*(1-))
    pi = prevalence
    denom = tpr*pi + fpr*(1-pi)
    return (tpr*pi / denom) if denom > 0 else np.nan

def expected_fp_per_10k(benign_rate, fpr):
    return fpr * benign_rate * 10000

def _scores_to_preds(scores, threshold=0.5):
    s = np.asarray(scores, dtype=float)
    return (s >= float(threshold)).astype(int)

def _threshold_for_fpr(y_true, y_score, target_fpr):
    """Find a threshold that achieves ~target_fpr; returns (threshold, realized_fpr, realized_tpr)."""
    fpr, tpr, thr = roc_curve(y_true, y_score)
    order = np.argsort(fpr)
    fpr, tpr, thr = fpr[order], tpr[order], thr[order]
    if target_fpr <= fpr[0]:
        t = float(thr[0])
    elif target_fpr >= fpr[-1]:
        t = float(thr[-1])
    else:
        i = np.searchsorted(fpr, target_fpr, side="right")
        x0, x1 = fpr[i-1], fpr[i]
        t0, t1 = thr[i-1], thr[i]
        w = (target_fpr - x0) / (x1 - x0) if x1 != x0 else 0.0
        t = t0 + (t1 - t0) * w
    # realized metrics at that threshold
    y_pred = (np.asarray(y_score) >= t).astype(int)
    tn, fp, fn, tp = confusion_matrix(y_true, y_pred).ravel()
    real_fpr = fp / (fp + tn) if (fp+tn)>0 else np.nan
    real_tpr = tp / (tp + fn) if (tp+fn)>0 else np.nan
    return float(t), float(real_fpr), float(real_tpr)
